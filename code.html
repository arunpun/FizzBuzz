The Code for APP


                        //Start function
function getValues() {
    //Get values from the page
    let fizzValue = document.getElementById("fizzValue").value;
    let buzzValue = document.getElementById("buzzValue").value;

    //Type casting
    fizzValue = parseInt(fizzValue);
    buzzValue = parseInt(buzzValue);

    //Validation
    if (Number.isInteger(fizzValue) && Number.isInteger(buzzValue)) {
        let fbArray = fizzBuzz(fizzValue, buzzValue);

        //Display data
        displayData(fbArray);
    } else {
        alert("Error: You must enter integers!");
    }
}

//Number or fizz or buzz?
function fizzBuzz(fizzValue, buzzValue) {
    let returnArray = [];

    //Loop from 1 to 100
    for (let i = 1; i <= 100; i++) {
        //Check to see if divisible by both
        if (i % fizzValue == 0 && i % buzzValue == 0) {
            returnArray.push("FizzBuzz");
            //Check to see if divisible by fizz value
        } else if (i % fizzValue == 0) {
            returnArray.push("Fizz");
            //Check to see if divisible by buzz value
        } else if (i % buzzValue == 0) {
            returnArray.push("Buzz");
        } else {
            returnArray.push(i);
        }
    }

    return returnArray;
}

The code is structured in three functions, getValues(), fizzBuzz() & displayData()

getValues()

The function is triggered by an event listener, document.getElementById("btnSubmit").addEventListener("click", getValues); The function then creates two variables, fizzValue & buzzValue whose values are passed respectively with the document.getElementById().value; method which are then converted into integers with the pareseInt() method. Inside a for loop, these parsed variables are then verified with the Number.isInteger() method. If both of the expression evaluates to true, the function creates an array
